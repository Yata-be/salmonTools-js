全ブキコンプ期待値計算

コンプできたブキの種類数とWaves数をもとに、あなたの引き運が「上振れ/下振れ」なのかどうか判定します。

Tips: ブキ全50種をコンプできるWave数の期待値は約280です。是非挑戦してみましょう！

import matplotlib.pyplot as plt Coef = [0.2 + i * 0.016 for i in range(51)] # 確率係数 Prob = [[0]*51] Prob[0][0] = 1 for i in range(700): newProb = [] for j in range(51): if j == 0: newProb.append(Prob[-1][j] * Coef[j]) else: newProb.append(Prob[-1][j] * Coef[j] + Prob[-1][j-1] * (1 - Coef[j-1])) Prob.append(newProb) def click(*ags, **kws): weapons = int(Element("weapons").element.value) if weapons < 0 or weapons > 50: pyscript.write("error", "ブキ種類数は0以上50以下で入力してください。") pyscript.write("print", "") pyscript.write("overview", "") pyscript.write("plot", "") return 0 waves = int(Element("waves").element.value) if waves < 1 or waves > 700: pyscript.write("error", "Wave数は1以上700以下で入力してください。") pyscript.write("print", "") pyscript.write("overview", "") pyscript.write("plot", "") return 0 pyscript.write("error", "") y = [] global Prob for i in range(701): s = sum(Prob[i][weapons:]) y.append(s) if s > 1 - 10**-5: break x = [i for i in range(len(y))] ans = y[waves] * 100 if ans >= 50: fortune = round((ans - 50) * 2) pyscript.write("print", "あなたの運は下振れしています。下振れ度：" + str(fortune)) else: fortune = round((50 - ans) * 2) pyscript.write("print", "あなたの運は上振れしています。上振れ度：" + str(fortune)) pyscript.write("overview", str(waves) + " Wavesで " + str(weapons) + " 種類以上コンプできる確率は "+ str(round(ans)) + "％ です。") fig, ax = plt.subplots() ax.plot(x, y) ax.hlines(y[waves], 0, len(y)) pyscript.write("plot", fig)
